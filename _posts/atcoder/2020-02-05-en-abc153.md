## B. Common Raccoon vs. Monster
```c
#include <stdio.h>

int main(void)
{
	int h, n, t=0;
	scanf("%d %d", &h, &n);
	while(n--)
	{
		int a;
		scanf("%d", &a);
		t+=a;
	}

	printf("%s\n", (t>=h) ? "Yes" : "No");
	return 0;
}
```

## C. Fennec vs. Monster
```cpp
#include <stdio.h>
#include <vector>
#include <algorithm>

using namespace std;

int main(void)
{
    int n, k;
    unsigned long long na = 0;
    scanf("%d %d", &n, &k);

    if (k >= n)
    {
        printf("0\n");
    }
    else
    {
        vector<int> hv;
        while (n--)
        {
            int h;
            scanf("%d", &h);
            hv.push_back(h);
        }

        sort(hv.begin(), hv.end());

        n = hv.size() - k;

        for (int i=0; i<n; ++i)
        {
            na += hv[i];
        }
        printf("%llu\n", na);
    }

    return 0;
}
```

## D. Caracal vs. Monster
```c
#include <stdio.h>

int main(void)
{
    unsigned long long h, c=0;
    scanf("%llu", &h);

    while (h>0)
    {
        h >>= 1;
        ++c;
    }

    printf("%llu\n", ((unsigned long long)(1) << c)-1);
    return 0;
}
```
